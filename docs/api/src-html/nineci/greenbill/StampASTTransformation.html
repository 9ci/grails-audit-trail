040            public void visit(ASTNode[] astNodes, SourceUnit sourceUnit) {
041    
042                    String createdByField = CONF.getProperty("stamp.audit.createdBy");
043                    String editedByField = CONF.getProperty("stamp.audit.editedBy");
044                    String editedDateField = CONF.getProperty("stamp.audit.editedDate");
045                    String createdDateField = CONF.getProperty("stamp.audit.createdDate");
046    
047                    for (ASTNode astNode : astNodes) {
048                            if (astNode instanceof ClassNode) {
049                                    ClassNode classNode = (ClassNode) astNode;
050                                    List<FieldNode>  fnlist = classNode.getFields();                  
051                                    //LOG.info("[Audit stamp ASTTransformation] Adding propertie [edited..created] to class [" + classNode.getName() + "]");
052                                    //System.out.println(classNode.getName() + " - [Audit stamp ASTTransformation] Adding propertie [edited..created]");
053                                    if(editedByField!=null){
054                                            classNode.addProperty(editedByField, Modifier.PUBLIC, new ClassNode(Long.class), new ConstantExpression(0), null, null);
055                                    }
056                                    if(createdByField!=null){
057                                            classNode.addProperty(createdByField, Modifier.PUBLIC, new ClassNode(Long.class), new ConstantExpression(0), null, null);
058                                    }
059                                    Expression now = new ConstructorCallExpression(new ClassNode(java.util.Date.class),MethodCallExpression.NO_ARGUMENTS);
060                                    if(createdDateField!=null){
061                                            classNode.addProperty(createdDateField, Modifier.PUBLIC, new ClassNode(java.util.Date.class), now, null, null);
062                                            addNullableConstraint(classNode,createdDateField);
063                                    }
064                                    if(editedDateField!=null){
065                                            classNode.addProperty(editedDateField, Modifier.PUBLIC, new ClassNode(java.util.Date.class), now, null, null);
066                                            addNullableConstraint(classNode,editedDateField);
067                                    }
068                                    
069                                    /****comment this out if you don't want our the custom ID and table name stuff****/
070                                    addTableAndIdMapping(classNode);
071    
072    
073                            }
074                    }
075            }

077            public void addTableAndIdMapping(ClassNode classNode){
078                    FieldNode closure = classNode.getDeclaredField("mapping");
079    
080                    if(closure!=null){
081                            boolean hasTable=hasFieldInClosure(closure,"table");
082                            boolean hasId=hasFieldInClosure(closure,"id");
083    
084                            ClosureExpression exp = (ClosureExpression)closure.getInitialExpression();
085                            BlockStatement block = (BlockStatement) exp.getCode();
086    
087                            //this just adds an s to the class name for the table if its not specified 
088                            Boolean pluralize = (Boolean)getMap(CO,"stamp.mapping.pluralTable");
089                            if(!hasTable && pluralize!=null && pluralize){
090                                    String tablename = GrailsClassUtils.getShortName(classNode.getName())+"s";
091                                    //LOG.info("Added new mapping to assign table: " + tablename);
092                                    MethodCallExpression tableMeth = new MethodCallExpression(
093                                            VariableExpression.THIS_EXPRESSION,
094                                            new ConstantExpression("table"),
095                                            new ArgumentListExpression(new ConstantExpression(tablename)) 
096                                            );
097                                    //block = (BlockStatement) exp.getCode();
098                                    block.addStatement(new ExpressionStatement(tableMeth));
099                                    //System.out.println(classNode.getName()+" - Added table mapping " + tablename );
100                            }
101                            //This adds the ID generator that we use for domian classes
102                            Map tableconf = (Map)getMap(CO,"stamp.mapping.id");
103                            if(!hasId && tableconf!=null){
104                                    NamedArgumentListExpression namedarg = new NamedArgumentListExpression();
105                                    if(tableconf.get("column") != null){
106                                            namedarg.addMapEntryExpression(new ConstantExpression("column"), new ConstantExpression(tableconf.get("column").toString()));
107                                    }
108                                    if(tableconf.get("generator") != null){
109                                            namedarg.addMapEntryExpression(new ConstantExpression("generator"), new ConstantExpression(tableconf.get("generator").toString()));
110                                    }
111                                    MethodCallExpression tableMeth = new MethodCallExpression(
112                                            VariableExpression.THIS_EXPRESSION,
113                                            new ConstantExpression("id"),
114                                            namedarg
115                                            );
116                                    //block = (BlockStatement) exp.getCode();
117                                    block.addStatement(new ExpressionStatement(tableMeth));
118                                    //System.out.println(classNode.getName() + " - Added ID mapping with "+ tableconf);
119                            }
120                    }
121                    //System.out.println(block.toString());
122            }

124            public void addNullableConstraint(ClassNode classNode,String fieldName){
125                    FieldNode closure = classNode.getDeclaredField("constraints");
126    
127                    if(closure!=null){
128    
129                            ClosureExpression exp = (ClosureExpression)closure.getInitialExpression();
130                            BlockStatement block = (BlockStatement) exp.getCode();
131    
132                            if(!hasFieldInClosure(closure,fieldName)){
133                                    NamedArgumentListExpression namedarg = new NamedArgumentListExpression();
134                                    namedarg.addMapEntryExpression(new ConstantExpression("nullable"), new ConstantExpression(true));
135                                    MethodCallExpression constExpr = new MethodCallExpression(
136                                            VariableExpression.THIS_EXPRESSION,
137                                            new ConstantExpression(fieldName),
138                                            namedarg
139                                            );
140                                    block.addStatement(new ExpressionStatement(constExpr));
141                                    //System.out.println(classNode.getName() + " - Added nullabel constraint for "+ fieldName);
142                            }
143                    }
144                    //System.out.println(block.toString());
145            }

149            public boolean hasFieldInClosure(FieldNode closure, String fieldName){
150                    if(closure != null){
151                            ClosureExpression exp = (ClosureExpression) closure.getInitialExpression();
152                            BlockStatement block = (BlockStatement) exp.getCode();
153                            List<Statement> ments = block.getStatements();
154                            for(Statement expstat : ments){
155                                    if(expstat instanceof ExpressionStatement && ((ExpressionStatement)expstat).getExpression() instanceof MethodCallExpression){
156                                            MethodCallExpression methexp = (MethodCallExpression)((ExpressionStatement)expstat).getExpression();
157                                            ConstantExpression conexp = (ConstantExpression)methexp.getMethod();
158                                            if(conexp.getValue().equals(fieldName)){
159                                                    return true;
160                                            }
161                                    }
162                            }
163                    }
164                    return false;
165            }

168            static public String getContents(File aFile) {
169                    //...checks on aFile are elided
170                    StringBuilder contents = new StringBuilder();
171    
172                    try {
173                            //use buffering, reading one line at a time
174                            //FileReader always assumes default encoding is OK!
175                            BufferedReader input =  new BufferedReader(new FileReader(aFile));
176                            try {
177                                    String line = null; 
178                                    while (( line = input.readLine()) != null){
179                                            contents.append(line);
180                                            contents.append(System.getProperty("line.separator"));
181                                    }
182                            }
183                            finally {
184                                    input.close();
185                            }
186                    }
187                    catch (IOException ex){
188                            ex.printStackTrace();
189                    }
190    
191                    return contents.toString();
192            }

194            static public Object getMap(Map configMap, String keypath) {
195                    String keys[] = keypath.split("\\.");
196                    Map map = configMap;
197                    for(String key : keys){
198                            Object val = map.get(key);
199                            if(val !=null){
200                                    //System.out.println("got a key for are " +key);
201                                    if(val instanceof Map){
202                                            map = (Map)map.get(key);
203                                    } else{
204                                            return val;
205                                    }
206                            }else{
207                                    return null;
208                            }
209                    }
210                    return map;     
211            }
